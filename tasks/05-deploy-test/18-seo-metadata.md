## Sitemap (`app/sitemap.ts`)

```ts
import { MetadataRoute } from 'next'

export default function sitemap(): MetadataRoute.Sitemap {
  return [
    { url: 'https://foresight-apps.vercel.app', lastModified: new Date() },
    { url: 'https://foresight-apps.vercel.app/dashboard', lastModified: new Date() },
    { url: 'https://foresight-apps.vercel.app/frame/usdc-vault', lastModified: new Date() },
  ]
}
```

## Robots (`app/robots.ts`)

```ts
import { MetadataRoute } from 'next'

export default function robots(): MetadataRoute.Robots {
  return {
    rules: { userAgent: '*', allow: '/' },
    sitemap: 'https://foresight-apps.vercel.app/sitemap.xml',
  }
}
```
